```go
// backend/internal/services/execution_service.go
package services

import (
    "context"
    "encoding/json"
    "fmt"
    "os/exec"
    "time"
    
    "github.com/hibiken/asynq"
    "github.com/rs/zerolog"
    "gorm.io/gorm"
    
    "vulun-scan-backend/internal/models"
)

type ExecutionService struct {
    db     *gorm.DB
    queue  *asynq.Client
    sseHub *SSEHub
    logger zerolog.Logger
}

func NewExecutionService() *ExecutionService {
    return &ExecutionService{
        db:     database.GetDB(),
        queue:  queue.GetClient(),
        sseHub: sse.GetHub(),
        logger: log.With().Str("service", "execution").Logger(),
    }
}

func (s *ExecutionService) StartExecution(toolID uint, args map[string]interface{}, timeoutSeconds *int) (*models.CommandExecution, error) {
    // 1. 查询工具配置
    var tool models.Tool
    if err := s.db.First(&tool, toolID).Error; err != nil {
        return nil, fmt.Errorf("工具不存在: %w", err)
    }
    
    if !tool.IsActive {
        return nil, fmt.Errorf("工具已禁用")
    }
    
    // 2. 创建执行记录
    execution := &models.CommandExecution{
        ToolID:    toolID,
        Status:    models.StatusQueued,
        Args:      s.marshalArgs(args),
        CreatedAt: time.Now(),
    }
    
    if err := s.db.Create(execution).Error; err != nil {
        return nil, fmt.Errorf("创建执行记录失败: %w", err)
    }
    
    // 3. 构建任务载荷
    payload := map[string]interface{}{
        "execution_id":    execution.ID,
        "tool_id":         toolID,
        "args":            args,
        "timeout_seconds": s.getTimeout(timeoutSeconds, tool.TimeoutSeconds),
    }
    
    payloadBytes, _ := json.Marshal(payload)
    
    // 4. 入队任务
    // 幂等键（可选）：放入 payload 以影响 Unique 判定
    if key, _ := args["idempotency_key"].(string); key != "" {
        payload["idempotency_key"] = key
    }
    task := asynq.NewTask("tool:execute", payloadBytes)
    opts := []asynq.Option{
        asynq.Queue("default"),
        asynq.MaxRetry(3),
        asynq.Timeout(time.Duration(payload["timeout_seconds"].(int)+30) * time.Second),
    }
    if _, ok := payload["idempotency_key"]; ok {
        opts = append(opts, asynq.Unique(10*time.Minute))
    }
    info, err := s.queue.Enqueue(task, opts...)
    if err != nil {
        s.db.Model(execution).Update("status", models.StatusFailed)
        return nil, fmt.Errorf("任务入队失败: %w", err)
    }
    
    s.logger.Info().
        Uint("execution_id", execution.ID).
        Uint("tool_id", toolID).
        Str("task_id", info.ID).
        Msg("执行任务已入队")
    
    return execution, nil
}

func (s *ExecutionService) CancelExecution(executionID uint) error {
    var execution models.CommandExecution
    if err := s.db.First(&execution, executionID).Error; err != nil {
        return fmt.Errorf("执行记录不存在: %w", err)
    }
    
    if execution.Status != models.StatusRunning && execution.Status != models.StatusQueued {
        return fmt.Errorf("无法取消已完成的任务")
    }
    
    // 更新数据库状态
    now := time.Now()
    updates := map[string]interface{}{
        "status":      models.StatusCanceled,
        "finished_at": &now,
    }
    
    if err := s.db.Model(&execution).Updates(updates).Error; err != nil {
        return fmt.Errorf("更新状态失败: %w", err)
    }
    
    // 发送取消事件
    s.sseHub.Publish(executionID, SSEEvent{
        Event:       "canceled",
        ExecutionID: executionID,
        Data: map[string]interface{}{
            "canceled_at": now.Format(time.RFC3339),
        },
    })
    
    s.logger.Info().Uint("execution_id", executionID).Msg("执行任务已取消")
    return nil
}

func (s *ExecutionService) marshalArgs(args map[string]interface{}) string {
    data, _ := json.Marshal(args)
    return string(data)
}

func (s *ExecutionService) getTimeout(custom *int, toolDefault int) int {
    if custom != nil && *custom > 0 {
        return *custom
    }
    if toolDefault > 0 {
        return toolDefault
    }
    return 300 // 默认5分钟
}
```

### 1.2 Worker 任务执行器
```go
// backend/internal/workers/tool_executor.go
package workers

import (
    "bufio"
    "bytes"
    "context"
    "encoding/json"
    "fmt"
    "io"
    "os"
    "os/exec"
    "strings"
    "syscall"
    "text/template"
    "time"
    
    "github.com/hibiken/asynq"
    "github.com/rs/zerolog"
    "vulun-scan-backend/internal/models"
)

type ToolExecutor struct {
    db     *gorm.DB
    sseHub *SSEHub
    logger zerolog.Logger
}

func NewToolExecutor() *ToolExecutor {
    return &ToolExecutor{
        db:     database.GetDB(),
        sseHub: sse.GetHub(),
        logger: log.With().Str("worker", "tool_executor").Logger(),
    }
}

func (w *ToolExecutor) ExecuteToolTask(ctx context.Context, t *asynq.Task) error {
    // 1. 解析任务参数
    var payload struct {
        ExecutionID    uint                   `json:"execution_id"`
        ToolID         uint                   `json:"tool_id"`
        Args           map[string]interface{} `json:"args"`
        TimeoutSeconds int                    `json:"timeout_seconds"`
    }
    
    if err := json.Unmarshal(t.Payload(), &payload); err != nil {
        return fmt.Errorf("解析任务参数失败: %w", err)
    }
    
    logger := w.logger.With().
        Uint("execution_id", payload.ExecutionID).
        Uint("tool_id", payload.ToolID).
        Logger()
    
    // 2. 查询工具和执行记录
    var tool models.Tool
    var execution models.CommandExecution
    
    if err := w.db.First(&tool, payload.ToolID).Error; err != nil {
        return fmt.Errorf("查询工具失败: %w", err)
    }
    
    if err := w.db.First(&execution, payload.ExecutionID).Error; err != nil {
        return fmt.Errorf("查询执行记录失败: %w", err)
    }
    
    // 检查是否已被取消
    if execution.Status == models.StatusCanceled {
        logger.Info().Msg("任务已被取消，跳过执行")
        return nil
    }
    
    // 3. 构建命令
    cmdArgs, err := w.buildCommand(tool.CommandTemplate, payload.Args)
    if err != nil {
        w.updateExecutionStatus(payload.ExecutionID, models.StatusFailed, nil, err.Error())
        return fmt.Errorf("构建命令失败: %w", err)
    }
    
    // 4. 执行命令
    return w.executeCommand(ctx, tool, payload.ExecutionID, cmdArgs, payload.TimeoutSeconds, logger)
}

func (w *ToolExecutor) executeCommand(ctx context.Context, tool models.Tool, executionID uint, cmdArgs []string, timeoutSeconds int, logger zerolog.Logger) error {
    // 创建超时上下文
    timeoutCtx, cancel := context.WithTimeout(ctx, time.Duration(timeoutSeconds)*time.Second)
    defer cancel()
    
    // 构建命令
    cmd := exec.CommandContext(timeoutCtx, cmdArgs[0], cmdArgs[1:]...)
    // 工作目录
    if strings.TrimSpace(tool.Workdir) != "" {
        cmd.Dir = tool.Workdir
    }
    // 环境变量合并
    if strings.TrimSpace(tool.Env) != "" {
        var envMap map[string]string
        if err := json.Unmarshal([]byte(tool.Env), &envMap); err == nil {
            envPairs := make([]string, 0, len(envMap))
            for k, v := range envMap {
                envPairs = append(envPairs, fmt.Sprintf("%s=%s", k, v))
            }
            cmd.Env = append(os.Environ(), envPairs...)
        }
    }
    // 创建进程组，便于整体终止
    cmd.SysProcAttr = &syscall.SysProcAttr{Setpgid: true}
    
    // 创建管道
    stdout, err := cmd.StdoutPipe()
    if err != nil {
        w.updateExecutionStatus(executionID, models.StatusFailed, nil, err.Error())
        return fmt.Errorf("创建stdout管道失败: %w", err)
    }
    
    stderr, err := cmd.StderrPipe()
    if err != nil {
        w.updateExecutionStatus(executionID, models.StatusFailed, nil, err.Error())
        return fmt.Errorf("创建stderr管道失败: %w", err)
    }
    
    // 启动命令
    startTime := time.Now()
    if err := cmd.Start(); err != nil {
        w.updateExecutionStatus(executionID, models.StatusFailed, nil, err.Error())
        return fmt.Errorf("启动命令失败: %w", err)
    }
    
    pid := cmd.Process.Pid
    logger.Info().Int("pid", pid).Strs("cmd", cmdArgs).Msg("命令已启动")
    
    // 更新状态为运行中
    w.updateExecutionStatus(executionID, models.StatusRunning, &pid, "")
    
    // 发送启动事件
    w.sseHub.Publish(executionID, SSEEvent{
        Event:       "started",
        ExecutionID: executionID,
        Data: map[string]interface{}{
            "pid":        pid,
            "started_at": startTime.Format(time.RFC3339),
            "command":    strings.Join(cmdArgs, " "),
        },
    })
    
    // 实时读取输出
    go w.streamOutput(executionID, stdout, "stdout")
    go w.streamOutput(executionID, stderr, "stderr")
    
    // 等待命令完成
    err = cmd.Wait()
    finishTime := time.Now()
    duration := finishTime.Sub(startTime)
    
    // 获取退出码
    exitCode := 0
    if err != nil {
        if exitError, ok := err.(*exec.ExitError); ok {
            exitCode = exitError.ExitCode()
        }
    }
    
    // 确定最终状态
    var finalStatus string
    var errorMessage string
    
    if timeoutCtx.Err() == context.DeadlineExceeded {
        finalStatus = models.StatusTimeout
        errorMessage = "执行超时"
    } else if exitCode == 0 {
        finalStatus = models.StatusCompleted
    } else {
        finalStatus = models.StatusFailed
        if err != nil {
            errorMessage = err.Error()
        }
    }
    
    // 更新最终状态
    w.updateFinalStatus(executionID, finalStatus, exitCode, errorMessage, finishTime)
    
    // 发送完成事件
    eventType := "completed"
    if finalStatus == models.StatusFailed {
        eventType = "failed"
    } else if finalStatus == models.StatusTimeout {
        eventType = "timeout"
    }
    
    w.sseHub.Publish(executionID, SSEEvent{
        Event:       eventType,
        ExecutionID: executionID,
        Data: map[string]interface{}{
            "exit_code":    exitCode,
            "finished_at":  finishTime.Format(time.RFC3339),
            "duration_ms":  duration.Milliseconds(),
            "error_message": errorMessage,
        },
    })
    
    logger.Info().
        Str("status", finalStatus).
        Int("exit_code", exitCode).
        Int64("duration_ms", duration.Milliseconds()).
        Msg("命令执行完成")
    
    return nil
}

func (w *ToolExecutor) streamOutput(executionID uint, pipe io.ReadCloser, streamType string) {
    defer pipe.Close()
    
    scanner := bufio.NewScanner(pipe)
    // 扩大单行缓冲，避免长日志被截断（默认64KB）
    scanner.Buffer(make([]byte, 0, 64*1024), 10*1024*1024)
    for scanner.Scan() {
        line := scanner.Text()
        
        // 发送日志事件
        w.sseHub.Publish(executionID, SSEEvent{
            Event:       "log",
            ExecutionID: executionID,
            Data: map[string]interface{}{
                "stream":    streamType,
                "timestamp": time.Now().Format(time.RFC3339),
                "line":      line,
            },
        })
    }
}

func (w *ToolExecutor) buildCommand(tpl string, args map[string]interface{}) ([]string, error) {
    // 优先支持 JSON 数组模板（推荐）：例如 ["nuclei", "-u", "{{.target}}", "-t", "{{.templates}}"]
    {
        var buf bytes.Buffer
        t, err := template.New("cmd").Option("missingkey=error").Parse(tpl)
        if err == nil {
            if err := t.Execute(&buf, args); err == nil {
                rendered := strings.TrimSpace(buf.String())
                if strings.HasPrefix(rendered, "[") {
                    var arr []string
                    if err := json.Unmarshal([]byte(rendered), &arr); err == nil {
                        return arr, nil
                    }
                }
                // 非数组模板：退化为基于空白分隔（注意：不支持带空格参数）
                return strings.Fields(rendered), nil
            }
        }
    }
    // 回退：原样分词（最低保真）
    return strings.Fields(tpl), nil
}

func (w *ToolExecutor) updateExecutionStatus(executionID uint, status string, pid *int, errorMessage string) {
    updates := map[string]interface{}{
        "status": status,
    }
    
    if pid != nil {
        updates["pid"] = *pid
        now := time.Now()
        updates["started_at"] = &now
    }
    
    if errorMessage != "" {
        updates["error_message"] = errorMessage
    }
    
    w.db.Model(&models.CommandExecution{}).Where("id = ?", executionID).Updates(updates)
}

func (w *ToolExecutor) updateFinalStatus(executionID uint, status string, exitCode int, errorMessage string, finishedAt time.Time) {
    updates := map[string]interface{}{
        "status":      status,
        "exit_code":   exitCode,
        "finished_at": &finishedAt,
    }
    
    if errorMessage != "" {
        updates["error_message"] = errorMessage
    }
    
    w.db.Model(&models.CommandExecution{}).Where("id = ?", executionID).Updates(updates)
}
```

### 1.3 SSE Hub 实现
```go
// backend/internal/sse/hub.go
package sse

import (
    "encoding/json"
    "fmt"
    "sync"
    "time"
)

type SSEEvent struct {
    Event       string      `json:"event"`
    ExecutionID uint        `json:"execution_id"`
    Data        interface{} `json:"data"`
}

type SSEHub struct {
    subscribers map[uint]map[chan []byte]bool // executionID -> channels
    seq         map[uint]uint64               // 每个执行的事件自增ID
    mu          sync.RWMutex
}

func NewSSEHub() *SSEHub {
    hub := &SSEHub{
        subscribers: make(map[uint]map[chan []byte]bool),
        seq:         make(map[uint]uint64),
    }
    // 启动清理协程
    go hub.cleanup()
    return hub
}

func (h *SSEHub) nextID(executionID uint) uint64 {
    h.mu.Lock()
    defer h.mu.Unlock()
    h.seq[executionID] = h.seq[executionID] + 1
    return h.seq[executionID]
}

func (h *SSEHub) Subscribe(executionID uint) chan []byte {
    h.mu.Lock()
    defer h.mu.Unlock()
    ch := make(chan []byte, 100) // 缓冲通道避免阻塞
    if h.subscribers[executionID] == nil {
        h.subscribers[executionID] = make(map[chan []byte]bool)
    }
    h.subscribers[executionID][ch] = true
    return ch
}

func (h *SSEHub) Unsubscribe(executionID uint, ch chan []byte) {
    h.mu.Lock()
    defer h.mu.Unlock()
    if subscribers, exists := h.subscribers[executionID]; exists {
        delete(subscribers, ch)
        close(ch)
        if len(subscribers) == 0 {
            delete(h.subscribers, executionID)
            delete(h.seq, executionID)
        }
    }
}

func (h *SSEHub) Publish(executionID uint, event SSEEvent) {
    h.mu.RLock()
    subscribers, exists := h.subscribers[executionID]
    h.mu.RUnlock()
    if !exists || len(subscribers) == 0 {
        return
    }
    // 序列化事件
    eventData, err := json.Marshal(event.Data)
    if err != nil {
        return
    }
    id := h.nextID(executionID)
    // 具名事件 + 事件ID + 数据 +（可选）重连时间
    message := fmt.Sprintf("id: %d\nevent: %s\ndata: %s\n\n", id, event.Event, eventData)
    messageBytes := []byte(message)
    // 广播
    h.mu.RLock()
    for ch := range subscribers {
        select {
        case ch <- messageBytes:
        default:
            // 通道已满，丢弃该消息以避免阻塞
        }
    }
    h.mu.RUnlock()
}

func (h *SSEHub) cleanup() {
    ticker := time.NewTicker(5 * time.Minute)
    defer ticker.Stop()
    for range ticker.C {
        h.mu.Lock()
        for executionID, subscribers := range h.subscribers {
            for ch := range subscribers {
                select {
                case ch <- []byte(": keepalive\n\n"):
                default:
                    delete(subscribers, ch)
                    close(ch)
                }
            }
            if len(subscribers) == 0 {
                delete(h.subscribers, executionID)
                delete(h.seq, executionID)
            }
        }
        h.mu.Unlock()
    }
}

var globalHub *SSEHub
var once sync.Once

func GetHub() *SSEHub {
    once.Do(func() { globalHub = NewSSEHub() })
    return globalHub
}
```

```go
// backend/internal/handlers/tool_execution.go
package handlers

import (
    "strconv"
    
    "github.com/gin-gonic/gin"
    "vulun-scan-backend/internal/models"
    "vulun-scan-backend/internal/services"
    "vulun-scan-backend/internal/utils/response"
)

// StartToolExecution 启动工具执行
// @Summary 启动工具执行
// @Description 启动指定工具的执行任务
// @Tags 工具执行
// @Accept json
// @Produce json
// @Param id path int true "工具ID"
// @Param request body models.StartExecutionRequest true "执行参数"
// @Success 200 {object} response.Response{data=models.StartExecutionResponseData}
// @Failure 400 {object} response.Response
// @Failure 500 {object} response.Response
// @Router /api/v1/tools/{id}/executions [post]
func StartToolExecution(c *gin.Context) {
    // 解析工具ID
    toolIDStr := c.Param("id")
    toolID, err := strconv.ParseUint(toolIDStr, 10, 32)
    if err != nil {
        response.ValidationErrorResponse(c, "无效的工具ID")
        return
    }
    
    // 绑定请求参数
    var req models.StartExecutionRequest
    if err := c.ShouldBindJSON(&req); err != nil {
        response.ValidationErrorResponse(c, "请求参数错误: "+err.Error())
        return
    }
    
    // 调用服务层
    executionService := services.NewExecutionService()
    execution, err := executionService.StartExecution(uint(toolID), req.Args, req.TimeoutSeconds)
    if err != nil {
        response.InternalServerErrorResponse(c, "启动执行失败: "+err.Error())
        return
    }
    
    // 构建响应
    responseData := models.StartExecutionResponseData{
        ExecutionID: execution.ID,
        Status:      execution.Status,
        CreatedAt:   execution.CreatedAt.Format("2006-01-02T15:04:05Z07:00"),
    }
    
    response.SuccessResponse(c, responseData)
}

// GetExecution 获取执行详情
// @Summary 获取执行详情
// @Description 获取指定执行任务的详细信息
// @Tags 工具执行
// @Produce json
// @Param id path int true "执行ID"
// @Success 200 {object} response.Response{data=models.ExecutionDetailResponseData}
// @Failure 404 {object} response.Response
// @Failure 500 {object} response.Response
// @Router /api/v1/executions/{id} [get]
func GetExecution(c *gin.Context) {
    executionIDStr := c.Param("id")
    executionID, err := strconv.ParseUint(executionIDStr, 10, 32)
    if err != nil {
        response.ValidationErrorResponse(c, "无效的执行ID")
        return
    }
    
    executionService := services.NewExecutionService()
    execution, err := executionService.GetExecution(uint(executionID))
    if err != nil {
        response.NotFoundResponse(c, "执行记录不存在")
        return
    }
    
    responseData := models.ExecutionDetailResponseData{
        ID:           execution.ID,
        ToolID:       execution.ToolID,
        Status:       execution.Status,
        ExitCode:     execution.ExitCode,
        Pid:          execution.Pid,
        ErrorMessage: execution.ErrorMessage,
        StartedAt:    formatTimePtr(execution.StartedAt),
        FinishedAt:   formatTimePtr(execution.FinishedAt),
        CreatedAt:    execution.CreatedAt.Format("2006-01-02T15:04:05Z07:00"),
        UpdatedAt:    execution.UpdatedAt.Format("2006-01-02T15:04:05Z07:00"),
    }
    
    response.SuccessResponse(c, responseData)
}

// StreamExecution SSE事件流
// @Summary 执行事件流
// @Description 通过SSE获取执行任务的实时事件流
// @Tags 工具执行
// @Produce text/event-stream
// @Param id path int true "执行ID"
// @Success 200 {string} string "SSE事件流"
// @Router /api/v1/executions/{id}/stream [get]
func StreamExecution(c *gin.Context) {
    executionIDStr := c.Param("id")
    executionID, err := strconv.ParseUint(executionIDStr, 10, 32)
    if err != nil {
        response.ValidationErrorResponse(c, "无效的执行ID")
        return
    }
    
    // 设置SSE响应头
    c.Header("Content-Type", "text/event-stream")
    c.Header("Cache-Control", "no-cache")
    c.Header("Connection", "keep-alive")
    c.Header("Access-Control-Allow-Origin", "*")
    
    // 订阅事件流
    sseHub := sse.GetHub()
    eventChan := sseHub.Subscribe(uint(executionID))
    defer sseHub.Unsubscribe(uint(executionID), eventChan)
    
    // 发送初始连接确认
    // 初始连接确认（带事件ID由 Hub 维护）
    c.Writer.Write([]byte("event: connected\ndata: {\"execution_id\":" + executionIDStr + "}\n\n"))
    c.Writer.Flush()
    
    // 持续发送事件
    for {
        select {
        case event, ok := <-eventChan:
            if !ok {
                return
            }
            
            c.Writer.Write(event)
            c.Writer.Flush()
            
        case <-c.Request.Context().Done():
            return
        }
    }
}

func formatTimePtr(t *time.Time) *string {
    if t == nil {
        return nil
    }
    formatted := t.Format("2006-01-02T15:04:05Z07:00")
    return &formatted
}
```
