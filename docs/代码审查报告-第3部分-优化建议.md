### 💡 4.1 减少重复代码 - 默认值设置

**问题：** 相同的分页默认值逻辑在多个 Handler 中重复

**解决方案：**
```go
// models/base.go
type BasePaginationRequest struct {
    Page      int    `form:"page"`
    PageSize  int    `form:"page_size"`
    SortBy    string `form:"sort_by"`
    SortOrder string `form:"sort_order"`
}

func (r *BasePaginationRequest) SetDefaults() {
    if r.Page <= 0 { r.Page = 1 }
    if r.PageSize <= 0 { r.PageSize = 10 }
    if r.SortBy == "" { r.SortBy = "updated_at" }
    if r.SortOrder == "" { r.SortOrder = "desc" }
}

// 使用
type GetOrgsRequest struct {
    BasePaginationRequest
}

func GetOrgs(c *gin.Context) {
    var req models.GetOrgsRequest
    c.ShouldBindQuery(&req)
    req.SetDefaults()  // ✅ 一行搞定
}
```

### 💡 4.2 优化日志 - 添加请求追踪

**解决方案：** 在 middleware 中为每个请求添加 request_id

```go
func Logger() gin.HandlerFunc {
    return func(c *gin.Context) {
        requestID := c.GetString("X-Request-ID")
        logger := log.With().Str("request_id", requestID).Logger()
        c.Set("logger", &logger)
        c.Next()
    }
}

// 在 Handler 中使用
func GetOrgs(c *gin.Context) {
    logger := c.MustGet("logger").(*zerolog.Logger)
    logger.Info().Msg("Fetching organizations")
}
```

    return func(c *gin.Context) {
        if c.Request.Method == "GET" {
            c.Header("Cache-Control", fmt.Sprintf("public, max-age=%d", int(maxAge.Seconds())))
        }
        c.Next()
    }
}

### 💡 4.6 前端错误边界

// components/error-boundary.tsx
export class ErrorBoundary extends React.Component<Props, State> {
  static getDerivedStateFromError(error: Error) {
    return { hasError: true, error }
  }
  
  render() {
    if (this.state.hasError) {
      return <ErrorFallback error={this.state.error} />
    }
    return this.props.children
  }
}

// 使用

### 💡 4.7 批量操作优化

**当前：** N 次事务  
**优化：** 1 次事务

```go
func (s *DomainService) BatchDeleteDomainsFromOrganization(req) (int, int, error) {
    return s.db.Transaction(func(tx *gorm.DB) error {
        // ✅ 批量删除关联
        result := tx.Where("organization_id = ? AND domain_id IN ?", req.OrgID, req.DomainIDs).
            Delete(&models.OrganizationDomain{})
        
        // ✅ 查找孤儿域名
        var orphanIDs []uint
        tx.Raw("SELECT id FROM domains WHERE id IN ? AND NOT EXISTS (...)", req.DomainIDs).Scan(&orphanIDs)
        
        // ✅ 批量删除孤儿
        if len(orphanIDs) > 0 {
            tx.Where("id IN ?", orphanIDs).Delete(&models.Domain{})
        }
        return nil
    })
}
```

func RateLimit(rps int) gin.HandlerFunc {
    limiter := rate.NewLimiter(rate.Limit(rps), rps*2)
    return func(c *gin.Context) {
        if !limiter.Allow() {
            c.JSON(429, gin.H{"error": "Too many requests"})
            c.Abort()
            return
        }
        c.Next()
    }
}

        start := time.Now()
        c.Next()

}
